new(1, 1);private string _currentMode = "adult";private readonly List<Action<string>> _observers = new();    public string CurrentMode => _currentMode;    public async Task<bool> SwitchModeAsync(string newMode, string password)    {        await _lock.WaitAsync();        try        {            // 验证管理员密码（哈希比对）            if (!ValidatePassword(password)) return false;         // 清空本地缓存（MemoryCache / Blazor状态）            ClearCache();                        return true;        }        finally { _lock.Release(); }    }        private bool
